    def __init__(self):
        super(NNLM, self).__init__()

        self.d = nn.Parameter(torch.randn(n_hidden).type(dtype))

        x = x.view(-1, n_steps * m)
        hidden_out = torch.tanh(torch.mm(x, self.H) + self.d)
        output = torch.mm(x, self.W) + torch.mm(hidden_out, self.U) + self.b
        return output
